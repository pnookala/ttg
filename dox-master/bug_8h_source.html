49 class MemoryWatchpoint_x86_64 {
50 public:
51 // x86 debugging registers are described in see
52 // https://www.intel.com/content/www/us/en/architecture-and-technology/64-ia-32-architectures-software-developer-system-programming-manual-325384.html
53 enum DebugRegister { kDR0 = 0, kDR1 = 1, kDR2 = 2, kDR3 = 3 };
54
55 enum Size {
56 kByte = 0, // 1 byte - 00
57 kHalfWord = 1, // 2 bytes - 01
58 kWord = 3, // 4 bytes - 11
59 // kDoubleWord = 2, // 8 bytes - 10 NOT SUPPORTED BY SOME CHIPS!
60 kSizeMask = 3 // mask 11
61 };
62
63 enum BreakState {
64 kDisabled = 0, // disabled - 00
65 kEnabledLocally = 1, // task local - 01
66 kEnabledGlobally = 2, // global - 10
67 kBreakStateMask = 3 // mask 11
68 };
69
70 enum Condition {
71 kWhenExecuted = 0, // on execution - 00
72 kWhenWritten = 1, // on write - 01
73 kWhenWrittenOrRead = 3, // on read or write - 11
74 kConditionMask = 3 // mask 11
75 };

103 Pool &set(void *addr, Size size, Condition cond, const pthread_t *thread) {
104 const auto it = pool_.find(thread);
105 assert(it != pool_.end());
106 // make sure there is no watchpoint for this address already
107 for (auto &watchpt_ptr : it->second) {
108 if (watchpt_ptr && watchpt_ptr->address() == addr) return *this;
109 }
110 // now create a watchpoint
111 for (auto dr = 0; dr != nwatchpoints_per_thread; ++dr) {
112 auto &watchpt_ptr = it->second[dr];
113 if (!watchpt_ptr) {
114 watchpt_ptr =
115 std::make_shared<MemoryWatchpoint_x86_64>(addr, size, cond, static_cast<DebugRegister>(dr), thread);
116 return *this;
117 }
118 }
119 return *this;
120 }
121
125 MemoryWatchpoint_x86_64 *find(void *addr, const pthread_t *thread) {
126 const auto it = pool_.find(thread);
127 assert(it != pool_.end());
128 for (auto &watchpt_ptr : it->second) {
129 if (watchpt_ptr && watchpt_ptr->address() == addr) return watchpt_ptr.get();
130 }
131 return nullptr;
132 }
133
135 Pool &clear(void *addr, const pthread_t *thread) {
136 const auto it = pool_.find(thread);
137 assert(it != pool_.end());
138 for (auto &watchpt_ptr : it->second) {
139 if (watchpt_ptr && watchpt_ptr->address() == addr) {
140 watchpt_ptr.reset();
141 return *this;
142 }
143 }
144 return *this;
145 }
146
147 private:
148 std::unordered_map<const pthread_t *, std::vector<std::shared_ptr<MemoryWatchpoint_x86_64>>> pool_;
149
155 explicit Pool(const std::vector<const pthread_t *> &threads) {
156 for (const auto &thread : threads) {
157 assert(thread != nullptr);
158 pool_[thread].resize(nwatchpoints_per_thread);
159 }
160 }
161
162 static std::shared_ptr<Pool> &get_instance() {
163 static std::shared_ptr<Pool> instance_;
164 return instance_;
165 }
166 };

196 inline uint64_t MakeFlags(DebugRegister reg, BreakState state, Condition cond, Size size) {
197 // N.B. each register takes 2 bits in DR7
198 return (state | cond << 16 | size << 24) << (2 * reg);
199 }
200
201 inline uint64_t MakeMask(DebugRegister reg) { return MakeFlags(reg, kBreakStateMask, kConditionMask, kSizeMask); }

205 void init(bool create) {
206#if defined(__APPLE__) && defined(__x86_64__)
207 x86_debug_state dr;
208 mach_msg_type_number_t dr_count = x86_DEBUG_STATE_COUNT;
209
210 mach_port_t target_mach_thread = pthread_mach_thread_np(*thread_);
211
212 kern_return_t rc =
213 thread_get_state(target_mach_thread, x86_DEBUG_STATE, reinterpret_cast<thread_state_t>(&dr), &dr_count);
214
215 if (create && rc != KERN_SUCCESS)
216 throw std::runtime_error("MemoryWatchpoint_x86_64::MemoryWatchpoint_x86_64(): thread_get_state failed");
217
218 switch (dr_) {
219 case kDR0:
220 dr.uds.ds64.__dr0 = reinterpret_cast<uint64_t>(addr_);
221 break;
222 case kDR1:
223 dr.uds.ds64.__dr1 = reinterpret_cast<uint64_t>(addr_);
224 break;
225 case kDR2:
226 dr.uds.ds64.__dr2 = reinterpret_cast<uint64_t>(addr_);
227 break;
228 case kDR3:
229 dr.uds.ds64.__dr3 = reinterpret_cast<uint64_t>(addr_);
230 break;
231 }
232
233 dr.uds.ds64.__dr7 &= ~MakeMask(dr_);
234
235 dr.uds.ds64.__dr7 |= MakeFlags(dr_, create ? kEnabledLocally : kDisabled, cond_, size_);
236
237 rc = thread_set_state(target_mach_thread, x86_DEBUG_STATE, reinterpret_cast<thread_state_t>(&dr), dr_count);
238
239 if (create && rc != KERN_SUCCESS)
240 throw std::runtime_error("MemoryWatchpoint_x86_64::MemoryWatchpoint_x86_64(): thread_set_state failed");
241#endif // defined(__APPLE__) && defined(__x86_64__)
242 }
243 };
