18 /* Wrapper allowing implementations to provide copies of data the user
19 * passed to send and broadcast. The value returned by operator() will
20 * be passed to all terminals. By default, the user-provided copy is
21 * returned.
22 * Implementations may provide specializations using the ttg::Runtime tag.
23 * TODO: can we avoid the three overloads?
24 */
25 template<ttg::Runtime Runtime>
26 struct value_copy_handler {
27 template<typename Value>
28 inline constexpr
29 Value&& operator()(Value&& value) const {
30 return std::forward<Value>(value);
31 }
32
33 template<typename Value>
34 inline constexpr
35 const Value& operator()(const Value& value) const {
36 return value;
37 }
38
39 template<typename Value>
40 inline constexpr
41 Value& operator()(Value& value) const {
42 return value;
43 }
44 };

156 void broadcast(const std::tuple<RangesT...>& keylists, valueT&& value, std::tuple<output_terminalsT...> &t) {
157 if constexpr (ttg::meta::is_iterable_v<std::tuple_element_t<KeyId, std::tuple<RangesT...>>>) {
158 if (std::distance(std::begin(std::get<KeyId>(keylists)), std::end(std::get<KeyId>(keylists))) > 0) {
159 std::get<i>(t).broadcast(std::get<KeyId>(keylists), value);
160 }
161 } else {
162 std::get<i>(t).broadcast(std::get<KeyId>(keylists), value);
163 }
164 if constexpr(sizeof...(I) > 0) {
165 detail::broadcast<KeyId+1, I...>(keylists, value, t);
166 }
