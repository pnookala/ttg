113 if (broadcast_callback) {
114 if constexpr (ttg::meta::is_iterable_v<rangeT>) {
115 broadcast_callback(ttg::span(&(*std::begin(keylist)), std::distance(std::begin(keylist), std::end(keylist))), value);
116 } else {

118 broadcast_callback(ttg::span<const keyT>(&keylist, 1), value);

121 if constexpr (ttg::meta::is_iterable_v<rangeT>) {
122 for (auto&& key : keylist) send(key, value);
123 } else {
124 /* single element */
125 send(keylist, value);
126 }
127 }

286 template <typename Key = keyT, typename Value = valueT>
287 std::enable_if_t<meta::is_none_void_v<Key,Value> && std::is_same_v<Value,std::remove_reference_t<Value>>,void>
288 send(const Key &key, Value &&value) {
289 std::size_t N = successors().size();
290 // find the first terminal that can consume the value
291 std::size_t move_terminal = N - 1;
292 for (std::size_t i = 0; i != N; ++i) {
293 if (successors().at(i)->get_type() == TerminalBase::Type::Consume) {
294 move_terminal = i;
295 break;
296 }
297 }
298 if (N > 0) {
299 // send copies to every terminal except the one we will move the results to
300 for (std::size_t i = 0; i != N; ++i) {
301 if (i != move_terminal) {
302 TerminalBase *successor = successors().at(i);
303 if (successor->get_type() == TerminalBase::Type::Read) {
304 static_cast<In<keyT, std::add_const_t<valueT>> *>(successor)->send(key, value);
305 } else if (successor->get_type() == TerminalBase::Type::Consume) {
306 static_cast<In<keyT, valueT> *>(successor)->send(key, value);
307 }
308 }
309 }
310 {
311 TerminalBase *successor = successors().at(move_terminal);
312 static_cast<In<keyT, valueT> *>(successor)->send(key, std::forward<Value>(value));
313 }
314 }
315 }
