180// Factory function to assist in wrapping a callable with signature
181//
182// case 1 (keyT != void): void op(const input_keyT&, std::tuple<input_valuesT&...>&&, std::tuple<output_terminalsT...>&)
183// case 2 (keyT == void): void op(std::tuple<input_valuesT&...>&&, std::tuple<output_terminalsT...>&)
184template <typename keyT, typename funcT, typename... input_valuesT, typename... output_edgesT>
185auto make_tt_tpl(funcT &&func, const std::tuple<ttg::Edge<keyT, input_valuesT>...> &inedges,
186 const std::tuple<output_edgesT...> &outedges, const std::string &name = "wrapper",
187 const std::vector<std::string> &innames = std::vector<std::string>(
188 std::tuple_size<std::tuple<ttg::Edge<keyT, input_valuesT>...>>::value, "input"),
189 const std::vector<std::string> &outnames =
190 std::vector<std::string>(std::tuple_size<std::tuple<output_edgesT...>>::value, "output")) {
191 using output_terminals_type = typename ttg::edges_to_output_terminals<std::tuple<output_edgesT...>>::type;
192
193 // TT needs actual types of arguments to func ... extract them and pass to CallableWrapTTArgs
194 // 1. func_args_t = {const input_keyT&, std::tuple<input_valuesT...>&&, std::tuple<output_terminalsT...>&}
195 using func_args_t = boost::callable_traits::args_t<funcT>;
196 constexpr const auto num_args = std::tuple_size<func_args_t>::value;
197 constexpr const auto void_key = ttg::meta::is_void_v<keyT>;
198 static_assert(num_args == (void_key ? 2 : 3),
199 "ttg::make_tt_tpl(func, ...): func must take 3 arguments (or 2, if keyT=void)");
200 // 2. input_args_t = {input_valuesT&&...}
201 using input_args_t = std::decay_t<typename std::tuple_element<void_key ? 0 : 1, func_args_t>::type>;
202 using decayed_input_args_t = ttg::meta::decayed_tuple_t<input_args_t>;
203 using wrapT = typename CallableWrapTTUnwrapTuple<funcT, keyT, output_terminals_type, input_args_t>::type;
204 // not sure if we need this level of type checking ...
205 // TODO determine the generic signature of func
206 if constexpr (!void_key) {
207 static_assert(
208 std::is_same_v<typename std::tuple_element<0, func_args_t>::type, const keyT &>,
209 "ttg::make_tt_tpl(func, inedges, outedges): first argument of func must be const keyT& (unless keyT = void)");
210 }
211 static_assert(std::is_same_v<decayed_input_args_t, std::tuple<input_valuesT...>>,
212 "ttg::make_tt_tpl(func, inedges, outedges): inedges value types do not match argument types of func");
213 static_assert(
214 std::is_same_v<typename std::tuple_element<num_args - 1, func_args_t>::type, output_terminals_type &>,
215 "ttg::make_tt_tpl(func, inedges, outedges): last argument of func must be std::tuple<output_terminals_type>&");
216
217 return std::make_unique<wrapT>(std::forward<funcT>(func), inedges, outedges, name, innames, outnames);
218}

220// Factory function to assist in wrapping a callable with signature
221//
222// case 1 (keyT != void): void op(const input_keyT&, input_valuesT&&..., std::tuple<output_terminalsT...>&)
223// case 2 (keyT == void): void op(input_valuesT&&..., std::tuple<output_terminalsT...>&)
224//
225// input edges can contain at most one control input, and it must be last, if present
226template <typename keyT, typename funcT, typename... input_edge_valuesT, typename... output_edgesT>
227auto make_tt(funcT &&func, const std::tuple<ttg::Edge<keyT, input_edge_valuesT>...> &inedges,
228 const std::tuple<output_edgesT...> &outedges, const std::string &name = "wrapper",
229 const std::vector<std::string> &innames = std::vector<std::string>(
230 std::tuple_size<std::tuple<ttg::Edge<keyT, input_edge_valuesT>...>>::value, "input"),
231 const std::vector<std::string> &outnames =
232 std::vector<std::string>(std::tuple_size<std::tuple<output_edgesT...>>::value, "output")) {
233 // ensure input types do not contain Void
234 static_assert(ttg::meta::is_none_Void_v<input_edge_valuesT...>, "ttg::Void is for internal use only, do not use it");
235 // ensure input types have at most 1 control input (and it is last, if present)
236 static_assert(ttg::meta::is_none_void_v<input_edge_valuesT...> || ttg::meta::is_last_void_v<input_edge_valuesT...>,
237 "at most one void input can be handled, and it must come last");
238
239 using output_terminals_type = typename ttg::edges_to_output_terminals<std::tuple<output_edgesT...>>::type;
240
241 // TT needs actual types of arguments to func ... extract them and pass to CallableWrapTTArgs
242 // 1. func_args_t = {const input_keyT&, input_valuesT&&..., std::tuple<output_terminalsT...>&}
243 using func_args_t = boost::callable_traits::args_t<funcT>;
244 constexpr auto num_args = std::tuple_size<func_args_t>::value;
245 constexpr auto void_key = ttg::meta::is_void_v<keyT>;
246 constexpr auto have_void_datum = !ttg::meta::is_none_void_v<input_edge_valuesT...>;
247
248 // if have data inputs and (always last) control input, convert last input to Void to make logic easier
249 using input_values_full_tuple_type = std::tuple<ttg::meta::void_to_Void_t<std::decay_t<input_edge_valuesT>>...>;
250 using input_values_tuple_type =
251 std::conditional_t<!have_void_datum, input_values_full_tuple_type,
252 typename ttg::meta::drop_last_n<input_values_full_tuple_type, std::size_t{1}>::type>;
253 // make sure that input_values_tuple_type does not contain void
254 static_assert(!ttg::meta::is_any_void_v<input_values_tuple_type>);
255
256 static_assert(num_args == std::tuple_size_v<input_values_tuple_type> + (void_key ? 1 : 2),
257 "ttg::make_tt(func, inedges): func's # of args != # of inedges");
258 // 2. input_args_t = {input_valuesT&&...}
259 using input_args_t = typename ttg::meta::take_first_n<
260 typename ttg::meta::drop_first_n<func_args_t, std::size_t(void_key ? 0 : 1)>::type,
261 std::tuple_size_v<func_args_t> - (void_key ? 1 : 2)>::type;
262 using decayed_input_args_t = ttg::meta::decayed_tuple_t<input_args_t>;
263 // 3. full_input_args_t = !have_void_datum ? input_args_t : input_args_t+void
264 using full_input_args_t =
265 std::conditional_t<!have_void_datum, input_args_t, ttg::meta::tuple_concat_t<input_args_t, std::tuple<void>>>;
266 using wrapT = typename CallableWrapTTArgsUnwrapTuple<funcT, keyT, output_terminals_type, full_input_args_t>::type;
267 // not sure if we need this level of type checking ...
268 // TODO determine the generic signature of func
269 if constexpr (!void_key) {
270 static_assert(
271 std::is_same_v<typename std::tuple_element<0, func_args_t>::type, const keyT &>,
272 "ttg::make_tt(func, inedges, outedges): first argument of func must be const keyT& (unless keyT = void)");
273 }
274 static_assert(std::is_same_v<decayed_input_args_t, input_values_tuple_type>,
275 "ttg::make_tt(func, inedges, outedges): inedges value types do not match argument types of func");
276 static_assert(
277 std::is_same_v<typename std::tuple_element<num_args - 1, func_args_t>::type, output_terminals_type &>,
278 "ttg::make_tt(func, inedges, outedges): last argument of func must be std::tuple<output_terminals_type>&");
279
280 return std::make_unique<wrapT>(std::forward<funcT>(func), inedges, outedges, name, innames, outnames);
281}
