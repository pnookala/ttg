256 int counter; // Tracks the number of arguments finalized
257 std::array<std::size_t, numins>

259 // for any type of input: 0 = finalized;
260 // for a streaming input the following values are possible:
261 // - 0: finalized
262 // - std::numeric_limits<std::size_t>::max(): initial state (there is no value yet)
263 // - 1: if streaming: have a value, waiting for more
264 // - n: if nonstreaming: expect this many more values

300 if constexpr (!ttg::meta::is_void_v<keyT> && !ttg::meta::is_empty_tuple_v<input_values_tuple_type>) {

309 derived->op(derived->output_terminals);

333 // there are 6 types of set_arg:
334 // - case 1: nonvoid Key, complete Value type
335 // - case 2: nonvoid Key, void Value, mixed (data+control) inputs
336 // - case 3: nonvoid Key, void Value, no inputs

342 // case 1:
343 template <std::size_t i, typename Key, typename Value>
344 std::enable_if_t<!ttg::meta::is_void_v<Key> && !std::is_void_v<std::decay_t<Value>>, void> set_arg(const Key &key,
345 Value &&value) {
346 using valueT = typename std::tuple_element<i, input_values_full_tuple_type>::type; // Should be T or const T
347 static_assert(std::is_same_v<std::decay_t<Value>, std::decay_t<valueT>>,
348 "TT::set_arg(key,value) given value of type incompatible with TT");
349
350 const auto owner = keymap(key);
351 if (owner != world.rank()) {
352 ttg::trace(world.rank(), ":", get_name(), " : ", key, ": forwarding setting argument : ", i);
353 // should be able on the other end to consume value (since it is just a temporary byproduct of serialization)
354 // BUT compiler vomits when const std::remove_reference_t<Value>& -> std::decay_t<Value>
355 // this exposes bad design in MemFuncWrapper (probably similar bugs elsewhere?) whose generic operator()
356 // should use memfun's argument types (since that's what will be called) rather than misautodeduce in a
357 // particular context P.S. another issue is in send_am which can execute both remotely (where one can always
358 // move arguments) and locally
359 // here we know that this will be a remove execution, so we prepare to take rvalues;
360 // send_am will need to separate local and remote paths to deal with this
361 worldobjT::send(owner, &ttT::template set_arg<i, Key, const std::remove_reference_t<Value> &>, key, value);
362 } else {
363 ttg::trace(world.rank(), ":", get_name(), " : ", key, ": received value for argument : ", i);
364
365 accessorT acc;
366 if (cache.insert(acc, key)) acc->second = new TTArgs(this->priomap(key)); // It will be deleted by the task q
367 TTArgs *args = acc->second;
368
369 if (args->nargs[i] == 0) {
370 ttg::print_error(world.rank(), ":", get_name(), " : ", key, ": error argument is already finalized : ", i);
371 throw std::runtime_error("TT::set_arg called for a finalized stream");
372 }
373
374 const auto &reducer = std::get<i>(input_reducers);
375 if (reducer) { // is this a streaming input? reduce the received value
376 // N.B. Right now reductions are done eagerly, without spawning tasks
377 // this means we must lock
378 args->lock();
379 if constexpr (!ttg::meta::is_void_v<valueT>) { // for data values
380 // have a value already? if not, set, otherwise reduce
381 if (args->nargs[i] == std::numeric_limits<std::size_t>::max()) {
382 this->get<i, std::decay_t<valueT> &>(args->input_values) = std::forward<Value>(value);
383
384 // now have a value, reset nargs
385 // check if we have a stream size for the op, which has precedence over the global setting.
386 if (args->stream_size[i] != 0) {
387 args->nargs[i] = args->stream_size[i];
388 } else if (static_streamsize[i] != 0) {
389 args->stream_size[i] = static_streamsize[i];
390 args->nargs[i] = static_streamsize[i];
391 } else {
392 args->nargs[i] = 1;
393 }
394 } else {
395 reducer(this->get<i, std::decay_t<valueT> &>(args->input_values), value);
396 }
397 } else {
398 reducer(); // even if this was a control input, must execute the reducer for possible side effects
399 }
400 // update the counter if the stream is bounded
401 // this assumes that the stream size is set before data starts flowing ... strong-typing streams will solve
402 // this
403 if (args->stream_size[i] != 0) {
404 args->nargs[i]--;
405 if (args->nargs[i] == 0) args->counter--;
406 }
407 args->unlock();
408 } else { // this is a nonstreaming input => set the value
409 if constexpr (!ttg::meta::is_void_v<valueT>) { // for data values
410 this->get<i, std::decay_t<valueT> &>(args->input_values) = std::forward<Value>(value);
411 }
412 args->nargs[i] = 0;
413 args->counter--;
414 }
415
416 // ready to run the task?
417 if (args->counter == 0) {
418 ttg::trace(world.rank(), ":", get_name(), " : ", key, ": submitting task for op ");
419 args->derived = static_cast<derivedT *>(this);
420 args->key = key;
421
422 using ttg::hash;
423 auto curhash = hash<keyT>{}(key);
424
425 if (curhash == threaddata.key_hash && threaddata.call_depth < 6) { // Needs to be externally configurable
426
427 // ttg::print("directly invoking:", get_name(), key, curhash, threaddata.key_hash, threaddata.call_depth);
428 ttT::threaddata.call_depth++;
429 if constexpr (!ttg::meta::is_void_v<keyT> && !ttg::meta::is_empty_tuple_v<input_values_tuple_type>) {
430 static_cast<derivedT *>(this)->op(key, args->make_input_refs(), output_terminals); // Runs immediately
431 } else if constexpr (!ttg::meta::is_void_v<keyT> && ttg::meta::is_empty_tuple_v<input_values_tuple_type>) {
432 static_cast<derivedT *>(this)->op(key, output_terminals); // Runs immediately
433 } else if constexpr (ttg::meta::is_void_v<keyT> && !ttg::meta::is_empty_tuple_v<input_values_tuple_type>) {
434 static_cast<derivedT *>(this)->op(args->make_input_refs(), output_terminals); // Runs immediately
435 } else if constexpr (ttg::meta::is_void_v<keyT> && ttg::meta::is_empty_tuple_v<input_values_tuple_type>) {
436 static_cast<derivedT *>(this)->op(output_terminals); // Runs immediately
437 } else
438 abort();
439 ttT::threaddata.call_depth--;
440
441 } else {
442 // ttg::print("enqueuing task", get_name(), key, curhash, threaddata.key_hash, threaddata.call_depth);
443 world.impl().impl().taskq.add(args);
444 }
445
446 cache.erase(acc);
447 }
448 }
449 }

795 // Copy/assign/move forbidden ... we could make it work using
796 // PIMPL for this base class. However, this instance of the base
797 // class is tied to a specific instance of a derived class a
798 // pointer to which is captured for invoking derived class
799 // functions. Thus, not only does the derived class have to be
800 // involved but we would have to do it in a thread safe way
801 // including for possibly already running tasks and remote
802 // references. This is not worth the effort ... wherever you are
803 // wanting to move/assign an TT you should be using a pointer.

809 // Registers the callback for the i'th input terminal
810 template <typename terminalT, std::size_t i>
811 void register_input_callback(terminalT &input) {
812 static_assert(std::is_same<keyT, typename terminalT::key_type>::value,
813 "TT::register_input_callback(terminalT) -- incompatible terminalT");
814 using valueT = std::decay_t<typename terminalT::value_type>;
815
817 // case 1: nonvoid key, nonvoid value
819 if constexpr (!ttg::meta::is_void_v<keyT> && !ttg::meta::is_empty_tuple_v<input_values_tuple_type> &&
820 !std::is_void_v<valueT>) {
821 auto move_callback = [this](const keyT &key, valueT &&value) {
822 set_arg<i, keyT, valueT>(key, std::forward<valueT>(value));
823 };
824 auto send_callback = [this](const keyT &key, const valueT &value) {
825 set_arg<i, keyT, const valueT &>(key, value);
826 };
827 auto setsize_callback = [this](const keyT &key, std::size_t size) { set_argstream_size<i>(key, size); };
828 auto finalize_callback = [this](const keyT &key) { finalize_argstream<i>(key); };
829 input.set_callback(send_callback, move_callback, {}, setsize_callback, finalize_callback);
830 }
832 // case 4: void key, nonvoid value
834 else if constexpr (ttg::meta::is_void_v<keyT> && !ttg::meta::is_empty_tuple_v<input_values_tuple_type> &&
835 !std::is_void_v<valueT>) {
836 auto move_callback = [this](valueT &&value) { set_arg<i, keyT, valueT>(std::forward<valueT>(value)); };
837 auto send_callback = [this](const valueT &value) { set_arg<i, keyT, const valueT &>(value); };
838 auto setsize_callback = [this](std::size_t size) { set_argstream_size<i>(size); };
839 auto finalize_callback = [this]() { finalize_argstream<i>(); };
840 input.set_callback(send_callback, move_callback, {}, setsize_callback, finalize_callback);
841 }
843 // case 2: nonvoid key, void value, mixed inputs
845 else if constexpr (!ttg::meta::is_void_v<keyT> && !ttg::meta::is_empty_tuple_v<input_values_tuple_type> &&
846 std::is_void_v<valueT>) {
847 auto send_callback = [this](const keyT &key) { set_arg<i, keyT, void>(key); };
848 auto setsize_callback = [this](const keyT &key, std::size_t size) { set_argstream_size<i>(key, size); };
849 auto finalize_callback = [this](const keyT &key) { finalize_argstream<i>(key); };
850 input.set_callback(send_callback, send_callback, {}, setsize_callback, finalize_callback);
851 }
853 // case 5: void key, void value, mixed inputs
855 else if constexpr (ttg::meta::is_all_void_v<keyT, valueT> &&
856 !ttg::meta::is_empty_tuple_v<input_values_tuple_type> && std::is_void_v<valueT>) {
857 auto send_callback = [this]() { set_arg<i, keyT, void>(); };
858 auto setsize_callback = [this](std::size_t size) { set_argstream_size<i>(size); };
859 auto finalize_callback = [this]() { finalize_argstream<i>(); };
860 input.set_callback(send_callback, send_callback, {}, setsize_callback, finalize_callback);
861 }
863 // case 3: nonvoid key, void value, no inputs
865 else if constexpr (!ttg::meta::is_void_v<keyT> && ttg::meta::is_empty_tuple_v<input_values_tuple_type> &&
866 std::is_void_v<valueT>) {
867 auto send_callback = [this](const keyT &key) { set_arg<keyT>(key); };
868 auto setsize_callback = [this](const keyT &key, std::size_t size) { set_argstream_size<i>(key, size); };
869 auto finalize_callback = [this](const keyT &key) { finalize_argstream<i>(key); };
870 input.set_callback(send_callback, send_callback, {}, setsize_callback, finalize_callback);
871 }
873 // case 6: void key, void value, no inputs
875 else if constexpr (ttg::meta::is_all_void_v<keyT, valueT> &&
876 ttg::meta::is_empty_tuple_v<input_values_tuple_type> && std::is_void_v<valueT>) {
877 auto send_callback = [this]() { set_arg<keyT>(); };
878 auto setsize_callback = [this](std::size_t size) { set_argstream_size<i>(size); };
879 auto finalize_callback = [this]() { finalize_argstream<i>(); };
880 input.set_callback(send_callback, send_callback, {}, setsize_callback, finalize_callback);
881 ttg::trace(world.rank(), ":", get_name(), " : set callbacks for terminal ", input.get_name(),
882 " assuming void {key,value} and no input");
883 } else
884 abort();
885 }
