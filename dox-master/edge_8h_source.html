16 template <typename keyT, typename valueT>
17 class Edge {
18 private:
19 // An EdgeImpl represents a single edge that most usually will
20 // connect a single output terminal with a single
21 // input terminal. However, we had to relax this constraint in
22 // order to easily accommodate connecting an input/output edge to
23 // an operation that to the outside looked like a single op but
24 // internally was implemented as multiple operations. Thus, the
25 // input/output edge has to connect to multiple terminals.
26 // Permitting multiple end points makes this much easier to
27 // compose, easier to implement, and likely more efficient at
28 // runtime. This is why outs/ins are vectors rather than pointers
29 // to a single terminal.
30 struct EdgeImpl {
31 std::string name;
32 std::vector<TerminalBase *> outs; // In<keyT, valueT> or In<keyT, const valueT>
33 std::vector<Out<keyT, valueT> *> ins;
34
35 EdgeImpl() : name(""), outs(), ins() {}
36
37 EdgeImpl(const std::string &name) : name(name), outs(), ins() {}
38
39 void set_in(Out<keyT, valueT> *in) {
40 if (ins.size()) {
41 trace("Edge: ", name, " : has multiple inputs");
42 }
43 ins.push_back(in);
44 try_to_connect_new_in(in);
45 }
46
47 void set_out(TerminalBase *out) {
48 if (outs.size()) {
49 trace("Edge: ", name, " : has multiple outputs");
50 }
51 outs.push_back(out);
52 try_to_connect_new_out(out);
53 }
54
55 void try_to_connect_new_in(Out<keyT, valueT> *in) const {
56 for (auto out : outs)
57 if (in && out) in->connect(out);
58 }
59
60 void try_to_connect_new_out(TerminalBase *out) const {
61 assert(out->get_type() != TerminalBase::Type::Write); // out must be an In<>
62 for (auto in : ins)
63 if (in && out) in->connect(out);
64 }
65
66 ~EdgeImpl() {
67 if (diagnose() && ((ins.size() == 0 && outs.size() != 0) || (ins.size() != 0 && outs.size() == 0))) {
68 print_error("Edge: destroying edge pimpl ('", name, "') with either in or out not assigned --- graph may be incomplete");
69 }
70 }
71 };
72
73 // We have a vector here to accommodate fusing multiple edges together
74 // when connecting them all to a single terminal.
75 mutable std::vector<std::shared_ptr<EdgeImpl>> p; // Need shallow copy semantics

88 template <typename... valuesT>
89 Edge(const Edge<keyT, valuesT> &... edges) : p(0) {
90 std::vector<Edge<keyT, valueT>> v = {edges...};
91 for (auto &edge : v) {
92 p.insert(p.end(), edge.p.begin(), edge.p.end());
93 }
94 }

97 bool live() const {
98 bool result = false;
99 for(const auto& edge: p) {
100 if (!edge->ins.empty())
101 return true;
102 }
103 return result;
104 }
